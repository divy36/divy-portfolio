import numpy as np
from scipy.stats import norm

def black_scholes(S, K, T, r, σ):
    d1 = (np.log(S/K) +
         (r + σ**2/2)*T) / (σ*np.sqrt(T))
    d2 = d1 - σ * np.sqrt(T)
    return S*norm.cdf(d1) - K*np.exp(
           -r*T)*norm.cdf(d2)
